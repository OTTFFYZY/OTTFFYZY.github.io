D. Alena And The Heater - Codeforces Round 466 (Div. 2)
Solutions Codeforces 贪心
Lastmod: 2025-01-03 周五 19:24:07

https://codeforces.com/contest/940/problem/D

题目大意

给出序列 $n(\le 10^5)$ 长的序列 $a$ 和 $b$。

$b_1 = b_2 = b_3 = b_4 = 0$。对于 $i \ge 5$

$b_i = 0$，如果 $a_{i - 4}, a_{i - 3}, a_{i - 2},a_{i - 1}, a_{i} > r$ 且 $b_{i - 4} = b_{i - 3} = b_{i - 2} = b_{i - 1} = 1$。

$b_i = 1$，如果 $a_{i - 4}, a_{i - 3}, a_{i - 2},a_{i - 1}, a_{i} < l$ 且 $b_{i - 4} = b_{i - 3} = b_{i - 2} = b_{i - 1} = 0$。

$b_i = b_{i - 1}$，其他情况。

给出一组可行的 $l, r$。输入保证有解。

简要题解

首先注意到，对于所有的 $b_{i - 1} \neq b_{i}$ 的位置，这里一定触发了 $b_i = 0$ 或 $b_i = 1$ 的条件，则首先 $l$ 或 $r$ 的值被这些位置限制。

考虑一连串 $11111$ 或 $00000$ 这样的情况，同样需要保证 $l, r$ 是合法的。

于是有

对于 $b[i - 4, i] = 00000$，$l \le \max_{i - 4}^i(a[i])$

对于 $b[i - 4, i] = 00001$，$l \ge \max_{i - 4}^i(a[i]) + 1$

对于 $b[i - 4, i] = 11110$，$r \le \min_{i - 4}^i(a[i]) - 1$

对于 $b[i - 4, i] = 11111$，$r \ge \min_{i - 4}^i(a[i])$

但是因为题目保证有合法的 $l, r$ 所以我们只需判断第 $2, 3$ 种情况，之后取最小的 $l$ 和最大的 $r$ 即可，不需要关注上界。

复杂度

$T$：$O(n)$

$S$：$O(n)$

代码实现

#include <bits/stdc++.h>

using LL = long long;

using PII = pair<int, int>;
using VI = vector<int>;
using MII = map<int, int>;

template<typename T> void cmin(T &x,const T &y) { if(y<x) x=y; }
template<typename T> void cmax(T &x,const T &y) { if(x<y) x=y; }
template<typename T> bool ckmin(T &x,const T &y) { 
    return y<x ? (x=y, true) : false; }
template<typename T> bool ckmax(T &x,const T &y) { 
    return x<y ? (x=y, true) : false; }
template<typename T> void cmin(T &x,T &y,const T &z) {// x<=y<=z 
    if(z<x) { y=x; x=z; } else if(z<y) y=z; }
template<typename T> void cmax(T &x,T &y,const T &z) {// x>=y>=z
    if(x<z) { y=x; x=z; } else if(y<z) y=z; }

// mt19937_64 rnd_64(chrono::system_clock::now().time_since_epoch().count());

/*
---------1---------2---------3---------4---------5---------6---------7---------
1234567890123456789012345678901234567890123456789012345678901234567890123456789
*/

const int INF = 1e9;

void solve() {
  int n; cin >> n;
  vector<int> a(n);
  for (int& i : a) cin >> i;
  string b; 
  cin >> b;

  vector<int> mi(n), ma(n);
  for (int i = 4; i < n; i++) {
    int nmi = INF, nma = -INF;
    for (int j = i - 4; j <= i; j++) {
      cmin(nmi, a[j]);
      cmax(nma, a[j]);
    }
    mi[i] = nmi;
    ma[i] = nma;
  }

  int l = -INF, r = INF;
  for (int i = 4; i < n; i++) {
    if (b[i] == b[i - 1]) continue;

    if (b[i] == '1') {
      cmax(l, ma[i] + 1);
    } else {
      cmin(r, mi[i] - 1);
    }
  }

  cout << l << ' ' << r << '\n'; 
}

int main() {
  int t = 1; 
  // cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}
