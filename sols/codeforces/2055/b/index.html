题目大意

给出 $n \ (n \le 2 \times 10^5)$ 长的两个数组 $a$ 和 $b$。$0 \le a_i, b_i \le 10^9$。

可以执行如下操作若干次：

将 $a_i$ 加 $1$，将除了 $a_i$ 以外的 $a$ 中元素减 $1$。

问是否可以对于所有的 $i$ 有 $a_i \ge b_i$。

简要题解

注意到，如果有两个位置 $a_i$ 是比 $b_i$ 小的，则增大其中一个位置必然会减小另一个则无法达成条件。

其他情况，如果只有一个 $a_i < b_i$ 则要看是否其他位置能提供足够的值去减即可。

复杂度

$T$：$O(n)$

$S$：$O(n)$

代码实现

#include <bits/stdc++.h>
using namespace std;

using LL = long long;
using ULL = unsigned long long;
using LD = long double;
using PII = pair<int, int>;
using VI = vector<int>;
using MII = map<int, int>;

template<typename T> void cmin(T &x,const T &y) { if(y<x) x=y; }
template<typename T> void cmax(T &x,const T &y) { if(x<y) x=y; }
template<typename T> bool ckmin(T &x,const T &y) { 
    return y<x ? (x=y, true) : false; }
template<typename T> bool ckmax(T &x,const T &y) { 
    return x<y ? (x=y, true) : false; }
template<typename T> void cmin(T &x,T &y,const T &z) {// x<=y<=z 
    if(z<x) { y=x; x=z; } else if(z<y) y=z; }
template<typename T> void cmax(T &x,T &y,const T &z) {// x>=y>=z
    if(x<z) { y=x; x=z; } else if(y<z) y=z; }

/*

const int INF = 0x3f3f3f3f;

void solve() {
  int n; cin >> n;
  vector<int> a(n), b(n);
  for (int& i : a) cin >> i;
  for (int& i : b) cin >> i;

  int mn = INF;
  int cnt = 0;
  int need = 0;
  for (int i = 0; i < n; i++) {
    if (a[i] >= b[i]) {
      cmin(mn, a[i] - b[i]);
    } else {
      cnt++;
      need = b[i] - a[i];
    }
  }

  if (cnt >= 2 || need > mn) {
    cout << "NO\n";
  } else {
    cout << "YES\n";
  }
}

int main() {
  int t = 1; 
  cin >> t;
  while (t--) {
    solve();
  }
  return 0;
}
